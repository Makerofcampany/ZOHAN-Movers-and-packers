import { useState, useEffect, FormEvent } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { Calculator, Check, ArrowRight, ShieldCheck, Mail, Phone, Calendar, User, FileText, CheckCircle2, RefreshCw } from 'lucide-react';
import { PRICING_PLANS, CITIES } from '../data';
import { QuoteRequest } from '../types';

interface PricingCalculatorProps {
  initialServiceType: string;
  onQuoteSubmitted: (newQuote: QuoteRequest) => void;
}

export default function PricingCalculator({ initialServiceType, onQuoteSubmitted }: PricingCalculatorProps) {
  const [activeTab, setActiveTab] = useState<'packages' | 'calculator'>('packages');
  
  // Calculator States
  const [originCity, setOriginCity] = useState('Karachi');
  const [destinationCity, setDestinationCity] = useState('Karachi');
  const [serviceType, setServiceType] = useState('House Shifting');
  const [homeSize, setHomeSize] = useState('2-bedroom');
  const [movingDate, setMovingDate] = useState('');
  const [hasPacking, setHasPacking] = useState(true);
  const [hasUnpacking, setHasUnpacking] = useState(true);
  const [loadingHelp, setLoadingHelp] = useState(true);
  const [needCarpenter, setNeedCarpenter] = useState(true);
  
  // User Contact States
  const [name, setName] = useState('');
  const [phone, setPhone] = useState('');
  const [email, setEmail] = useState('');
  const [message, setMessage] = useState('');
  
  // Form Wizard State
  const [wizardStep, setWizardStep] = useState(1); // 1: Estimate, 2: Contacts, 3: Success
  const [liveEstimate, setLiveEstimate] = useState(0);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [submittedQuoteId, setSubmittedQuoteId] = useState('');

  // Handle service pre-selection from parent
  useEffect(() => {
    if (initialServiceType) {
      setServiceType(initialServiceType);
      setActiveTab('calculator');
      setWizardStep(1);
    }
  }, [initialServiceType]);

  // Recalculate Live Shifting Price
  useEffect(() => {
    let price = 12000; // Base Local Local Shifting

    // Adjust by Shifting/Home Size
    switch (homeSize) {
      case '1-room': price = 10000; break;
      case '1-bedroom': price = 15000; break;
      case '2-bedroom': price = 24000; break;
      case '3-4-bedroom': price = 38000; break;
      case 'large-villa': price = 55000; break;
      case 'office-small': price = 20000; break;
      case 'office-large': price = 50000; break;
    }

    // Intercity surcharge
    const isIntercity = originCity !== destinationCity;
    if (isIntercity) {
      // Base Intercity overhead is high in Pakistan
      price += 45000;
      
      // Additional route estimates
      if (
        (originCity === 'Karachi' && destinationCity === 'Islamabad') ||
        (originCity === 'Islamabad' && destinationCity === 'Karachi')
      ) {
        price += 35000; // Multi-province diesel tax
      } else if (
        (originCity === 'Karachi' && destinationCity === 'Lahore') ||
        (originCity === 'Lahore' && destinationCity === 'Karachi')
      ) {
        price += 25000;
      } else if (
        (originCity === 'Lahore' && destinationCity === 'Islamabad') ||
        (originCity === 'Islamabad' && destinationCity === 'Lahore')
      ) {
        price += 12000;
      } else {
        price += 18000; // average other cities
      }
    }

    // Add-ons
    if (hasPacking) price += homeSize.includes('bedroom') ? 6000 : 3000;
    if (hasUnpacking) price += homeSize.includes('bedroom') ? 4000 : 2000;
    if (loadingHelp) price += homeSize.includes('large') || homeSize.includes('villa') ? 8000 : 4000;
    if (needCarpenter) price += 3500; // Standard carpenter service in PKR

    setLiveEstimate(price);
  }, [originCity, destinationCity, homeSize, hasPacking, hasUnpacking, loadingHelp, needCarpenter]);

  const loadPackageIntoCalculator = (pkgName: string) => {
    if (pkgName.includes('Standard')) {
      setHomeSize('1-bedroom');
      setOriginCity('Karachi');
      setDestinationCity('Karachi');
      setHasPacking(false);
      setHasUnpacking(false);
      setNeedCarpenter(true);
    } else if (pkgName.includes('Premium')) {
      setHomeSize('2-bedroom');
      setOriginCity('Karachi');
      setDestinationCity('Karachi');
      setHasPacking(true);
      setHasUnpacking(true);
      setNeedCarpenter(true);
    } else {
      setHomeSize('3-4-bedroom');
      setOriginCity('Karachi');
      setDestinationCity('Lahore');
      setHasPacking(true);
      setHasUnpacking(true);
      setNeedCarpenter(true);
    }
    setActiveTab('calculator');
    setWizardStep(1);
    
    // Scroll to calculator
    const el = document.getElementById('calculator-section-anchor');
    if (el) {
      el.scrollIntoView({ behavior: 'smooth' });
    }
  };

  const handleSubmitQuote = (e: FormEvent) => {
    e.preventDefault();
    if (!name || !phone || !movingDate) {
      alert('Please fill in your Name, Phone Number, and Moving Date.');
      return;
    }

    setIsSubmitting(true);

    // Simulate server side submission
    setTimeout(() => {
      const uniqueId = 'ZHN-' + Math.floor(100000 + Math.random() * 900000);
      const newRequest: QuoteRequest = {
        id: uniqueId,
        name,
        phone,
        email: email || 'No Email Provided',
        originCity,
        destinationCity,
        serviceType,
        movingDate,
        homeSize,
        hasPacking,
        hasUnpacking,
        loadingHelp,
        message: message || 'Urgent shifting quote request.',
        status: 'pending',
        estimatedPrice: liveEstimate,
        createdAt: new Date().toLocaleDateString('en-US', {
          year: 'numeric',
          month: 'long',
          day: 'numeric',
          hour: '2-digit',
          minute: '2-digit'
        })
      };

      // Call parent callback & update local storage
      onQuoteSubmitted(newRequest);
      
      setSubmittedQuoteId(uniqueId);
      setIsSubmitting(false);
      setWizardStep(3);
    }, 1200);
  };

  const resetCalculator = () => {
    setName('');
    setPhone('');
    setEmail('');
    setMessage('');
    setWizardStep(1);
  };

  return (
    <section id="pricing" className="py-20 md:py-28 bg-white text-slate-800 scroll-mt-12">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        
        {/* Section Header */}
        <div id="calculator-section-anchor" className="text-center max-w-3xl mx-auto mb-12 md:mb-16">
          <div className="text-orange-500 font-extrabold text-xs uppercase tracking-widest mb-2 flex items-center justify-center gap-2">
            <span className="w-3 h-0.5 bg-orange-500"></span>
            <span>Flexible Packages</span>
            <span className="w-3 h-0.5 bg-orange-500"></span>
          </div>
          <h2 className="text-3xl sm:text-4xl font-extrabold text-blue-950 font-display">
            Transparent Pricing, Zero Hidden Costs
          </h2>
          <p className="text-slate-500 text-sm mt-3">
            Choose a quick flat-rate package or build your custom relocation quote using our interactive real-time cost estimator.
          </p>

          {/* Tab Selection */}
          <div className="inline-flex p-1.5 bg-slate-100 rounded-2xl mt-8 border border-slate-200">
            <button
              onClick={() => setActiveTab('packages')}
              className={`px-6 py-2.5 rounded-xl text-sm font-bold transition-all focus:outline-none cursor-pointer ${
                activeTab === 'packages'
                  ? 'bg-blue-950 text-white shadow-md'
                  : 'text-slate-600 hover:text-blue-950'
              }`}
            >
              Flat-Rate Packages
            </button>
            <button
              onClick={() => setActiveTab('calculator')}
              className={`px-6 py-2.5 rounded-xl text-sm font-bold transition-all focus:outline-none cursor-pointer flex items-center gap-1.5 ${
                activeTab === 'calculator'
                  ? 'bg-blue-950 text-white shadow-md'
                  : 'text-slate-600 hover:text-blue-950'
              }`}
            >
              <Calculator className="w-4 h-4" />
              <span>Live Quote Estimator</span>
            </button>
          </div>
        </div>

        {/* Content Tabs */}
        <AnimatePresence mode="wait">
          {activeTab === 'packages' ? (
            <motion.div
              key="packages"
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              exit={{ opacity: 0, y: -15 }}
              transition={{ duration: 0.3 }}
              className="grid grid-cols-1 md:grid-cols-3 gap-8"
            >
              {PRICING_PLANS.map((plan) => (
                <div
                  key={plan.id}
                  className={`rounded-3xl p-8 text-left flex flex-col justify-between relative transition-all duration-300 ${
                    plan.popular
                      ? 'bg-blue-950 text-white border-2 border-orange-500 shadow-2xl scale-102 z-10'
                      : 'bg-slate-50 text-slate-800 border border-slate-200/80 hover:shadow-lg'
                  }`}
                >
                  {plan.popular && (
                    <span className="absolute -top-4 left-1/2 -translate-x-1/2 px-4 py-1.5 bg-orange-500 text-white text-[11px] font-black uppercase tracking-widest rounded-full shadow-lg">
                      Most Popular Plan
                    </span>
                  )}
                  <div>
                    <h3 className={`text-xl font-extrabold font-display ${plan.popular ? 'text-white' : 'text-blue-950'}`}>
                      {plan.name}
                    </h3>
                    <p className={`text-xs mt-2 leading-relaxed ${plan.popular ? 'text-slate-300' : 'text-slate-500'}`}>
                      {plan.description}
                    </p>
                    
                    <div className="my-6">
                      <span className={`text-2xl font-black font-display ${plan.popular ? 'text-orange-400' : 'text-blue-950'}`}>
                        {plan.priceLabel}
                      </span>
                      <span className="text-xs opacity-60 block mt-1">Est. base fare (Subject to distance)</span>
                    </div>

                    <div className={`h-px w-full my-6 ${plan.popular ? 'bg-white/10' : 'bg-slate-200'}`}></div>

                    <ul className="space-y-3.5">
                      {plan.features.map((feature, i) => (
                        <li key={i} className="flex items-start gap-2.5 text-xs md:text-sm">
                          <span className={`p-0.5 rounded-full mt-0.5 shrink-0 ${
                            plan.popular ? 'bg-orange-500/10 text-orange-400 border border-orange-500/20' : 'bg-blue-50 text-blue-900 border border-blue-100'
                          }`}>
                            <Check className="w-3.5 h-3.5 font-bold" />
                          </span>
                          <span className={plan.popular ? 'text-slate-200' : 'text-slate-600'}>{feature}</span>
                        </li>
                      ))}
                    </ul>
                  </div>

                  <div className="pt-8">
                    <button
                      onClick={() => loadPackageIntoCalculator(plan.name)}
                      className={`w-full py-3.5 font-bold text-sm rounded-xl transition-all active:scale-95 cursor-pointer ${
                        plan.popular
                          ? 'bg-orange-500 hover:bg-orange-600 text-white shadow-xl shadow-orange-500/25'
                          : 'bg-blue-950 hover:bg-blue-900 text-white'
                      }`}
                    >
                      Calculate With This Plan
                    </button>
                  </div>
                </div>
              ))}
            </motion.div>
          ) : (
            <motion.div
              key="calculator"
              initial={{ opacity: 0, y: 15 }}
              animate={{ opacity: 1, y: 0 }}
              exit={{ opacity: 0, y: -15 }}
              transition={{ duration: 0.3 }}
              className="bg-slate-50 rounded-3xl border border-slate-200/60 p-6 md:p-10 shadow-xl max-w-5xl mx-auto"
            >
              {/* Wizard Steps indicator */}
              <div className="flex items-center justify-center space-x-4 mb-8">
                <div className="flex items-center space-x-2">
                  <span className={`w-8 h-8 rounded-full flex items-center justify-center text-xs font-bold ${
                    wizardStep >= 1 ? 'bg-blue-950 text-white' : 'bg-slate-200 text-slate-500'
                  }`}>
                    1
                  </span>
                  <span className="text-xs font-bold text-slate-600 hidden sm:inline">Configure Move</span>
                </div>
                <div className={`h-0.5 w-12 ${wizardStep >= 2 ? 'bg-blue-950' : 'bg-slate-200'}`}></div>
                <div className="flex items-center space-x-2">
                  <span className={`w-8 h-8 rounded-full flex items-center justify-center text-xs font-bold ${
                    wizardStep >= 2 ? 'bg-blue-950 text-white' : 'bg-slate-200 text-slate-500'
                  }`}>
                    2
                  </span>
                  <span className="text-xs font-bold text-slate-600 hidden sm:inline">Contact Information</span>
                </div>
                <div className={`h-0.5 w-12 ${wizardStep >= 3 ? 'bg-blue-950' : 'bg-slate-200'}`}></div>
                <div className="flex items-center space-x-2">
                  <span className={`w-8 h-8 rounded-full flex items-center justify-center text-xs font-bold ${
                    wizardStep >= 3 ? 'bg-emerald-600 text-white' : 'bg-slate-200 text-slate-500'
                  }`}>
                    ✓
                  </span>
                  <span className="text-xs font-bold text-slate-600 hidden sm:inline">Confirmation</span>
                </div>
              </div>

              {/* Wizard content rendering */}
              <AnimatePresence mode="wait">
                {wizardStep === 1 && (
                  <motion.div
                    key="step-1"
                    initial={{ opacity: 0, x: 10 }}
                    animate={{ opacity: 1, x: 0 }}
                    exit={{ opacity: 0, x: -10 }}
                    className="grid grid-cols-1 lg:grid-cols-12 gap-10 text-left"
                  >
                    {/* Form Controls Left */}
                    <div className="lg:col-span-7 space-y-5">
                      <h4 className="text-lg font-bold text-blue-950 mb-3 font-display border-b border-slate-200 pb-2">
                        Relocation Parameters
                      </h4>

                      {/* Cities Origin / Destination */}
                      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                        <div>
                          <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Origin City</label>
                          <select
                            value={originCity}
                            onChange={(e) => setOriginCity(e.target.value)}
                            className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                          >
                            {CITIES.map((c) => <option key={c} value={c}>{c}</option>)}
                          </select>
                        </div>
                        <div>
                          <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Destination City</label>
                          <select
                            value={destinationCity}
                            onChange={(e) => setDestinationCity(e.target.value)}
                            className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                          >
                            {CITIES.map((c) => <option key={c} value={c}>{c}</option>)}
                          </select>
                        </div>
                      </div>

                      {/* Shifting Type & Home Size */}
                      <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                        <div>
                          <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Service Type</label>
                          <select
                            value={serviceType}
                            onChange={(e) => setServiceType(e.target.value)}
                            className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                          >
                            <option value="House Shifting">House Shifting</option>
                            <option value="Office Relocation">Office Relocation</option>
                            <option value="Furniture Moving">Furniture Moving</option>
                            <option value="Intercity Moving">Intercity Moving</option>
                            <option value="Warehouse Storage">Warehouse Storage</option>
                          </select>
                        </div>
                        <div>
                          <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Size / Inventory Volume</label>
                          <select
                            value={homeSize}
                            onChange={(e) => setHomeSize(e.target.value)}
                            className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                          >
                            <option value="1-room">Single Room / Chota Shifting</option>
                            <option value="1-bedroom">1 Bedroom Apartment/Portion</option>
                            <option value="2-bedroom">2 Bedroom Apartment/House</option>
                            <option value="3-4-bedroom">3-4 Bedroom Large House</option>
                            <option value="large-villa">Large Villa / Bungalow (5+ Beds)</option>
                            <option value="office-small">Small Office Shifting (1-10 Desks)</option>
                            <option value="office-large">Large Corporate Office (15+ Desks)</option>
                          </select>
                        </div>
                      </div>

                      {/* Moving Date */}
                      <div>
                        <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Target Moving Date</label>
                        <input
                          type="date"
                          value={movingDate}
                          onChange={(e) => setMovingDate(e.target.value)}
                          min={new Date().toISOString().split('T')[0]}
                          className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                          required
                        />
                      </div>

                      {/* Premium Add-ons Checkboxes */}
                      <div className="pt-2">
                        <label className="block text-xs font-bold text-slate-500 uppercase mb-3">Include Additional Services</label>
                        <div className="grid grid-cols-1 sm:grid-cols-2 gap-3">
                          <button
                            type="button"
                            onClick={() => setHasPacking(!hasPacking)}
                            className={`p-3.5 border rounded-xl flex items-center justify-between text-left transition-all focus:outline-none cursor-pointer ${
                              hasPacking
                                ? 'bg-orange-500/5 border-orange-500 text-blue-950 font-bold'
                                : 'bg-white border-slate-200 text-slate-600'
                            }`}
                          >
                            <span className="text-xs">Premium Packing Materials</span>
                            <span className={`w-4 h-4 rounded-full border flex items-center justify-center ${
                              hasPacking ? 'bg-orange-500 border-orange-500 text-white' : 'border-slate-300'
                            }`}>
                              {hasPacking && <span className="text-[9px]">✓</span>}
                            </span>
                          </button>

                          <button
                            type="button"
                            onClick={() => setHasUnpacking(!hasUnpacking)}
                            className={`p-3.5 border rounded-xl flex items-center justify-between text-left transition-all focus:outline-none cursor-pointer ${
                              hasUnpacking
                                ? 'bg-orange-500/5 border-orange-500 text-blue-950 font-bold'
                                : 'bg-white border-slate-200 text-slate-600'
                            }`}
                          >
                            <span className="text-xs">Unpacking & Arrangements</span>
                            <span className={`w-4 h-4 rounded-full border flex items-center justify-center ${
                              hasUnpacking ? 'bg-orange-500 border-orange-500 text-white' : 'border-slate-300'
                            }`}>
                              {hasUnpacking && <span className="text-[9px]">✓</span>}
                            </span>
                          </button>

                          <button
                            type="button"
                            onClick={() => setLoadingHelp(!loadingHelp)}
                            className={`p-3.5 border rounded-xl flex items-center justify-between text-left transition-all focus:outline-none cursor-pointer ${
                              loadingHelp
                                ? 'bg-orange-500/5 border-orange-500 text-blue-950 font-bold'
                                : 'bg-white border-slate-200 text-slate-600'
                            }`}
                          >
                            <span className="text-xs">Loader Boys / ground staff</span>
                            <span className={`w-4 h-4 rounded-full border flex items-center justify-center ${
                              loadingHelp ? 'bg-orange-500 border-orange-500 text-white' : 'border-slate-300'
                            }`}>
                              {loadingHelp && <span className="text-[9px]">✓</span>}
                            </span>
                          </button>

                          <button
                            type="button"
                            onClick={() => setNeedCarpenter(!needCarpenter)}
                            className={`p-3.5 border rounded-xl flex items-center justify-between text-left transition-all focus:outline-none cursor-pointer ${
                              needCarpenter
                                ? 'bg-orange-500/5 border-orange-500 text-blue-950 font-bold'
                                : 'bg-white border-slate-200 text-slate-600'
                            }`}
                          >
                            <span className="text-xs">Carpenter (Beds, Wardrobes)</span>
                            <span className={`w-4 h-4 rounded-full border flex items-center justify-center ${
                              needCarpenter ? 'bg-orange-500 border-orange-500 text-white' : 'border-slate-300'
                            }`}>
                              {needCarpenter && <span className="text-[9px]">✓</span>}
                            </span>
                          </button>
                        </div>
                      </div>
                    </div>

                    {/* Estimator Display Right */}
                    <div className="lg:col-span-5 flex flex-col justify-between">
                      <div className="bg-blue-950 text-white p-6 md:p-8 rounded-2xl relative overflow-hidden flex-1 border border-blue-900 shadow-xl flex flex-col justify-between">
                        <div>
                          <span className="px-2.5 py-1 bg-orange-500 text-white text-[9px] font-black uppercase tracking-wider rounded-md">
                            Zohan Movers Instant Quote
                          </span>
                          <h4 className="text-lg font-extrabold font-display mt-4">Estimated Shifting Fare</h4>
                          <p className="text-slate-300 text-xs mt-1">Real-time valuation based on custom parameters.</p>

                          <div className="my-8">
                            <span className="text-3xl sm:text-4xl font-black text-transparent bg-clip-text bg-gradient-to-r from-orange-400 to-orange-500 font-display">
                              PKR {liveEstimate.toLocaleString('en-US')}
                            </span>
                            <span className="text-xs text-slate-400 block mt-1">Estimated Shifting Toll + Labor</span>
                          </div>

                          <div className="border-t border-white/10 pt-4 space-y-2 text-xs text-slate-300">
                            <div className="flex justify-between">
                              <span>Distance Fee:</span>
                              <span>{originCity === destinationCity ? 'Local Shifting Flat' : 'Intercity Long Route'}</span>
                            </div>
                            <div className="flex justify-between">
                              <span>Shifting Size:</span>
                              <span className="capitalize">{homeSize.replace('-', ' ')}</span>
                            </div>
                            <div className="flex justify-between">
                              <span>Packaging Support:</span>
                              <span>{hasPacking ? 'Yes (Materials Included)' : 'Self Packing'}</span>
                            </div>
                            <div className="flex justify-between">
                              <span>Carpenter Required:</span>
                              <span>{needCarpenter ? 'Yes (On Site Expert)' : 'No'}</span>
                            </div>
                          </div>
                        </div>

                        <div className="mt-8 pt-4 border-t border-white/10 text-[11px] text-slate-400 leading-normal flex items-start gap-1.5">
                          <span className="text-orange-500 font-bold shrink-0">Note:</span>
                          <span>The quoted price is an automated realistic baseline. Final figures will be confirmed by our surveyor to guarantee absolute accuracy.</span>
                        </div>
                      </div>

                      <div className="pt-4 flex items-center justify-end">
                        <button
                          type="button"
                          onClick={() => setWizardStep(2)}
                          className="w-full py-4 bg-orange-500 hover:bg-orange-600 active:scale-95 text-white font-bold rounded-xl shadow-lg shadow-orange-500/25 flex items-center justify-center gap-2 cursor-pointer"
                        >
                          <span>Proceed to Contact Info</span>
                          <ArrowRight className="w-4 h-4" />
                        </button>
                      </div>
                    </div>
                  </motion.div>
                )}

                {wizardStep === 2 && (
                  <motion.form
                    key="step-2"
                    onSubmit={handleSubmitQuote}
                    initial={{ opacity: 0, x: 10 }}
                    animate={{ opacity: 1, x: 0 }}
                    exit={{ opacity: 0, x: -10 }}
                    className="grid grid-cols-1 lg:grid-cols-12 gap-10 text-left"
                  >
                    {/* User Contacts Left */}
                    <div className="lg:col-span-7 space-y-5">
                      <h4 className="text-lg font-bold text-blue-950 mb-3 font-display border-b border-slate-200 pb-2 flex items-center justify-between">
                        <span>Customer Shifting Details</span>
                        <button
                          type="button"
                          onClick={() => setWizardStep(1)}
                          className="text-xs text-orange-500 hover:underline font-bold"
                        >
                          ← Change Move Parameters
                        </button>
                      </h4>

                      <div>
