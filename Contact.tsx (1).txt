import { useState, FormEvent } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { Phone, Mail, MapPin, Send, CheckCircle2, MessageSquare, RefreshCw } from 'lucide-react';

interface ContactProps {
  onInquirySubmitted: (inquiry: { name: string; phone: string; message: string; date: string }) => void;
}

export default function Contact({ onInquirySubmitted }: ContactProps) {
  // Form state
  const [name, setName] = useState('');
  const [phone, setPhone] = useState('');
  const [email, setEmail] = useState('');
  const [message, setMessage] = useState('');
  
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [isSuccess, setIsSuccess] = useState(false);

  const handleFormSubmit = (e: FormEvent) => {
    e.preventDefault();
    if (!name || !phone || !message) {
      alert('Please fill in Name, Phone, and your Message.');
      return;
    }

    setIsSubmitting(true);

    // Simulate submission
    setTimeout(() => {
      onInquirySubmitted({
        name,
        phone,
        message,
        date: new Date().toLocaleDateString('en-US', {
          month: 'short',
          day: 'numeric',
          year: 'numeric',
          hour: '2-digit',
          minute: '2-digit'
        })
      });
      
      setIsSubmitting(false);
      setIsSuccess(true);
      
      // Reset
      setName('');
      setPhone('');
      setEmail('');
      setMessage('');
    }, 1000);
  };

  return (
    <section id="contact" className="py-20 md:py-28 bg-white overflow-hidden scroll-mt-8">
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        
        {/* Section Header */}
        <div className="text-center max-w-3xl mx-auto mb-16 md:mb-20">
          <div className="text-orange-500 font-extrabold text-xs uppercase tracking-widest mb-2 flex items-center justify-center gap-2">
            <span className="w-3 h-0.5 bg-orange-500"></span>
            <span>Get In Touch</span>
            <span className="w-3 h-0.5 bg-orange-500"></span>
          </div>
          <h2 className="text-3xl sm:text-4xl font-extrabold text-blue-950 font-display">
            Contact Zohan Movers & Packers
          </h2>
          <p className="text-slate-500 text-sm mt-3">
            Ready to plan your move? Reach out via our quick message form, call us directly, or chat on WhatsApp for 24/7 support.
          </p>
        </div>

        {/* Info & Form Split */}
        <div className="grid grid-cols-1 lg:grid-cols-12 gap-12 items-start">
          
          {/* Info Details & Map - Left */}
          <div className="lg:col-span-5 space-y-8 text-left">
            
            <div className="space-y-6">
              <h3 className="text-xl font-extrabold text-blue-950 font-display">Contact Information</h3>
              <p className="text-slate-500 text-sm">
                Our main office is situated on Jinnah Highway in Karachi. Feel free to visit us or contact our support lines for on-site survey bookings.
              </p>

              {/* Direct Details list */}
              <div className="space-y-4">
                <a
                  href="tel:03122800519"
                  className="flex items-center gap-4 p-4 bg-slate-50 rounded-2xl border border-slate-200/50 hover:border-orange-200 hover:bg-orange-500/[0.02] transition-colors group"
                >
                  <div className="w-10 h-10 rounded-xl bg-orange-100 flex items-center justify-center text-orange-500 group-hover:scale-105 transition-transform">
                    <Phone className="w-5 h-5" />
                  </div>
                  <div>
                    <span className="text-[10px] font-bold text-slate-400 uppercase tracking-widest block">Phone Helpline</span>
                    <span className="text-sm font-bold text-blue-950">+92 312 2800519</span>
                  </div>
                </a>

                <a
                  href="mailto:zohanmovers@gmail.com"
                  className="flex items-center gap-4 p-4 bg-slate-50 rounded-2xl border border-slate-200/50 hover:border-orange-200 hover:bg-orange-500/[0.02] transition-colors group"
                >
                  <div className="w-10 h-10 rounded-xl bg-orange-100 flex items-center justify-center text-orange-500 group-hover:scale-105 transition-transform">
                    <Mail className="w-5 h-5" />
                  </div>
                  <div>
                    <span className="text-[10px] font-bold text-slate-400 uppercase tracking-widest block">Email Address</span>
                    <span className="text-sm font-bold text-blue-950">zohanmovers@gmail.com</span>
                  </div>
                </a>

                <div className="flex items-center gap-4 p-4 bg-slate-50 rounded-2xl border border-slate-200/50">
                  <div className="w-10 h-10 rounded-xl bg-orange-100 flex items-center justify-center text-orange-500">
                    <MapPin className="w-5 h-5" />
                  </div>
                  <div>
                    <span className="text-[10px] font-bold text-slate-400 uppercase tracking-widest block">Office Location</span>
                    <span className="text-sm font-bold text-blue-950">Jinnah Highway, Karachi, Pakistan</span>
                  </div>
                </div>
              </div>
            </div>

            {/* Quick action buttons */}
            <div className="grid grid-cols-2 gap-3 pt-2">
              <a
                href="tel:03122800519"
                className="py-3 px-4 bg-blue-950 hover:bg-blue-900 text-white font-bold text-xs sm:text-sm rounded-xl text-center shadow flex items-center justify-center gap-1.5"
              >
                <Phone className="w-4 h-4" />
                <span>Click To Call</span>
              </a>

              <a
                href="https://wa.me/923122800519"
                target="_blank"
                rel="noopener noreferrer"
                className="py-3 px-4 bg-emerald-600 hover:bg-emerald-700 text-white font-bold text-xs sm:text-sm rounded-xl text-center shadow flex items-center justify-center gap-1.5"
              >
                {/* WhatsApp logo */}
                <svg className="w-4 h-4 fill-current" viewBox="0 0 24 24">
                  <path d="M12.031 6.172c-2.002 0-3.882.977-5.031 2.613-.679.967-1.031 2.115-1.031 3.298 0 1.228.384 2.417 1.11 3.411L6 20l4.639-1.218c.95.518 2.016.792 3.102.792 3.528 0 6.399-2.871 6.399-6.399 0-3.527-2.871-6.403-6.11-6.403h.001zm0 11.666a5.21 5.21 0 0 1-2.658-.726l-.19-.111-2.736.718.73-2.666-.123-.196a5.216 5.216 0 0 1-.803-2.796c0-2.884 2.348-5.232 5.232-5.232 2.885 0 5.233 2.348 5.233 5.232.001 2.884-2.348 5.233-5.233 5.233h-.002z"/>
                </svg>
                <span>WhatsApp Us</span>
              </a>
            </div>

            {/* Google Map Embed */}
            <div className="rounded-2xl overflow-hidden border-2 border-slate-150 h-52 shadow-sm relative">
              <iframe
                title="Zohan Movers Karachi Office Location"
                src="https://maps.google.com/maps?q=Jinnah%20Highway,%20Karachi,%20Pakistan&t=&z=13&ie=UTF8&iwloc=&output=embed"
                className="absolute inset-0 w-full h-full border-0"
                allowFullScreen
                loading="lazy"
                referrerPolicy="no-referrer-when-downgrade"
              ></iframe>
            </div>

          </div>

          {/* Contact Message Form - Right */}
          <div className="lg:col-span-7 bg-slate-50 p-6 sm:p-10 rounded-3xl border border-slate-200/50 shadow-sm text-left">
            <h3 className="text-xl font-extrabold text-blue-950 font-display mb-2">Send Us a Direct Message</h3>
            <p className="text-slate-500 text-xs sm:text-sm mb-6">
              Fill out your details and our team will get in touch with you shortly.
            </p>

            <AnimatePresence mode="wait">
              {!isSuccess ? (
                <motion.form
                  key="contact-form"
                  onSubmit={handleFormSubmit}
                  className="space-y-4"
                  initial={{ opacity: 0 }}
                  animate={{ opacity: 1 }}
                  exit={{ opacity: 0 }}
                >
                  <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
                    <div>
                      <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Your Name</label>
                      <input
                        type="text"
                        value={name}
                        onChange={(e) => setName(e.target.value)}
                        placeholder="e.g., Muhammad Ali"
                        className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                        required
                      />
                    </div>
                    <div>
                      <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Phone Number</label>
                      <input
                        type="tel"
                        value={phone}
                        onChange={(e) => setPhone(e.target.value)}
                        placeholder="e.g., 03122800519"
                        className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                        required
                      />
                    </div>
                  </div>

                  <div>
                    <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Email Address (Optional)</label>
                    <input
                      type="email"
                      value={email}
                      onChange={(e) => setEmail(e.target.value)}
                      placeholder="e.g., mail@example.com"
                      className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                    />
                  </div>

                  <div>
                    <label className="block text-xs font-bold text-slate-500 uppercase mb-1.5">Message / Shifting Requirements</label>
                    <textarea
                      value={message}
                      onChange={(e) => setMessage(e.target.value)}
                      rows={5}
                      placeholder="e.g., I want to shift my household goods from Gulshan-e-Iqbal to DHA Phase 5. Kindly provide a estimate quote."
                      className="w-full px-4 py-3 bg-white border border-slate-200 rounded-xl text-sm font-semibold focus:border-orange-500 focus:ring-1 focus:ring-orange-500 focus:outline-none"
                      required
                    />
                  </div>

                  <button
                    type="submit"
                    disabled={isSubmitting}
                    className="w-full py-4 bg-orange-500 hover:bg-orange-600 active:scale-95 text-white font-bold rounded-xl shadow-lg shadow-orange-500/20 transition-all flex items-center justify-center gap-2 cursor-pointer"
                  >
                    {isSubmitting ? (
                      <>
                        <RefreshCw className="w-5 h-5 animate-spin" />
                        <span>Sending Message...</span>
                      </>
                    ) : (
                      <>
                        <span>Send Message</span>
                        <Send className="w-4 h-4" />
                      </>
                    )}
                  </button>
                </motion.form>
              ) : (
                <motion.div
                  key="success-form"
                  initial={{ opacity: 0, scale: 0.95 }}
                  animate={{ opacity: 1, scale: 1 }}
                  exit={{ opacity: 0, scale: 0.95 }}
                  className="py-12 text-center space-y-4"
                >
                  <div className="w-16 h-16 bg-emerald-50 rounded-full flex items-center justify-center mx-auto border border-emerald-500 text-emerald-600">
                    <CheckCircle2 className="w-10 h-10" />
                  </div>
                  <h4 className="text-xl font-bold text-blue-950 font-display">Message Received!</h4>
                  <p className="text-sm text-slate-500 max-w-md mx-auto leading-relaxed">
                    Thank you for reaching out. Your message has been sent directly to the Zohan Movers support team. We will contact you at your phone number shortly.
                  </p>
                  <button
                    onClick={() => setIsSuccess(false)}
                    className="mt-4 px-5 py-2.5 bg-blue-950 hover:bg-blue-900 text-white font-semibold text-xs rounded-lg cursor-pointer"
                  >
                    Send Another Message
                  </button>
                </motion.div>
              )}
            </AnimatePresence>
          </div>

        </div>

      </div>
    </section>
  );
}
